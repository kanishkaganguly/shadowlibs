111  template<typename U>
112  struct printer {
113  static void print_body(const U &c, ostream_type &stream) {
114  using std::begin;
115  using std::end;
116 
117  auto it = begin(c);
118  const auto the_end = end(c);
119 
120  if (it != the_end) {
121  for (;;) {
122  stream << *it;
123 
124  if (++it == the_end) break;
125 
126  if (delimiters_type::values.delimiter != NULL)

128  }
129  }
